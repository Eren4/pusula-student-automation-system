CREATE TABLE student(
	student_id SERIAL PRIMARY KEY,
	student_name VARCHAR(255),
	student_surname VARCHAR(255),
	student_email VARCHAR(255) UNIQUE,
	student_password VARCHAR(255)
);

CREATE TABLE teacher(
	teacher_id SERIAL PRIMARY KEY,
	teacher_name VARCHAR(255),
	teacher_surname VARCHAR(255),
	teacher_email VARCHAR(255) UNIQUE,
	teacher_password VARCHAR(255)
);

CREATE TABLE admin(
 	admin_id SERIAL PRIMARY KEY,
 	admin_nickname VARCHAR(255),
 	admin_email VARCHAR(255) UNIQUE,
 	admin_password VARCHAR(255)
);

CREATE TABLE course(
 	course_id SERIAL PRIMARY KEY,
 	course_name VARCHAR(255),
 	teacher_id INT REFERENCES teacher(teacher_id),
 	course_ongoing BOOLEAN
);

CREATE TABLE grade(
 	course_id INT REFERENCES course(course_id) NOT NULL,
	student_id INT REFERENCES student(student_id) NOT NULL,
	student_grade NUMERIC,
	student_absence INT,
	student_comment VARCHAR(255),
	PRIMARY KEY (course_id, student_id)
);
